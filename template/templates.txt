#获取yaml文件中配置的数据
def getRequestData(file: str) -> list:
    if file.endswith('.yaml') or file.endswith('.yml'):
        rs = doYaml.get_data(os.path.join(DATA_PATH, file))
        url = URL + rs["path"]
        method = rs["method"]
        data = rs["data"]
        header = rs["header"]
        return [(url, method, data, header)]
    else:
        raise Exception("文件名只支持yaml或者yml")
